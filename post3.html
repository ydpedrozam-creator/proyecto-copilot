def DFS_recursivo(grafo, nodo, visitados=None):

def DFS_iterativo(grafo, inicio):
    """Recorre el grafo usando DFS de forma iterativa"""
    visitados = set()
    pila = [inicio]
    resultado = []
    
    while pila:
        nodo = pila.pop()  # Desapilar
        
        if nodo not in visitados:
            visitados.add(nodo)
            resultado.append(nodo)

            # Apilar todos los vecinos

                if vecino not in visitados:
                    pila.append(vecino)
    
    return resultado
